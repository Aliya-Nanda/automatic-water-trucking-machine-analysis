PARAMETER INPUT (DATA NYATA PROYEK KELOMPOK 2) ======================================================================== % -------------------- Data Geometri & Fisik -------------------- platform_size = [25, 20]; % cm (Panjang x Lebar chassis) m_mdf_structure = 0.85; % kg (Estimasi berat bodi laser-cut 6mm MDF) m_electronics = 0.40; % kg (Arduino Mega, 4 motor, baterai LiPo, sensor) m_motor = 0.12; % kg (4 motor DC @ 30g each) m_roda = 0.08; % kg (4 roda + as) % -------------------- Data Muatan (Payload) -------------------- n_cups = 4; % Jumlah gelas yang digunakan vol_per_cup = 200; % ml (Kapasitas standar gelas plastik) fill_percentage = 0.85; % 85% terisi (untuk menghindari tumpah) % -------------------- Data Arena & Misi -------------------- track_length = 3.0; % meter (Jarak tempuh satu arah) avg_velocity = 0.15; % m/s (Kecepatan cruising estimasi) docking_time = 10; % detik (Waktu proses pengisian air) bump_height = 20; % mm (Tinggi rintangan gundukan) bump_width = 40; % mm (Lebar rintangan = 20 x 2) n_bumps = 3; % Jumlah gundukan di arena % -------------------- Target Kinerja (Demands & Wishes) -------------------- target_water_volume = 550; % ml (DEMAND 1: Minimal 550ml) target_weight_ratio = 2; % (WISHES 1: Rasio berat < 2:1) target_mission_time = 60; % detik (WISHES 3: Misi < 60 detik) target_stability = 1.0; % cm (Batas simpangan vertikal aman) ======================================================================== 2. PERHITUNGAN TEKNIS (KPI CALCULATIONS) ======================================================================== % -------------------- Perhitungan Massa -------------------- total_water_vol = n_cups * vol_per_cup * fill_percentage; % ml m_water = total_water_vol / 1000; % kg (1L = 1kg) m_dry = m_mdf_structure + m_electronics + m_motor + m_roda; % Berat kosong m_total = m_dry + m_water; % Berat total % -------------------- Perhitungan Rasio Berat (WISHES 1) -------------------- weight_to_payload_ratio = m_dry / m_water; % -------------------- Perhitungan Waktu Misi (WISHES 3) -------------------- time_travel = (track_length * 2) / avg_velocity; % Maju + Mundur total_mission_time = time_travel + docking_time; % -------------------- Perhitungan Stabilitas (Getaran) -------------------- k_chassis = 800; % N/m (kekakuan struktural MDF) wn = sqrt(k_chassis / m_total); % Frekuensi natural (rad/s) % Variasi kecepatan untuk pengujian v_test = [0.1, 0.15, 0.2, 0.25, 0.3]; % m/s y_max = zeros(1, length(v_test)); % Informasi gundukan yang diperbarui fprintf('--- Informasi Gundukan ---\n'); fprintf(' Tinggi gundukan : %d mm\n', bump_height); fprintf(' Lebar gundukan : %d mm (20 x 2)\n', bump_width); fprintf(' Jumlah gundukan : %d buah\n', n_bumps); fprintf(' Total lintasan : %.0f mm (40 x 3)\n', bump_width * n_bumps); fprintf(' Jarak antar : %.0f mm\n\n', (track_length*1000 - bump_width*n_bumps)/(n_bumps-1)); for i = 1:length(v_test) % Waktu kontak dengan satu gundukan t_contact = (bump_width/1000) / v_test(i); % Frekuensi eksitasi dari gundukan f_excitation = 1 / t_contact; % Rasio frekuensi (r = f_excitation / fn) r = (f_excitation) / (wn/(2*pi)); % Faktor amplifikasi (untuk sistem underdamped/) zeta_assumed = 0.3; % Asumsi redaman rendah (tanpa suspensi aktif) amplification = 1 / sqrt((1 - r^2)^2 + (2*zeta_assumed*r)^2); % Simpangan maksimum yang terjadi (konversi mm ke cm) y_max(i) = (bump_height/10) * amplification; end % Cek apakah stabil (simpangan < batas aman) stability_status = y_max <= target_stability; % -------------------- Perhitungan Biaya (Tambahan) -------------------- cost_mdf = 50000; % Rp (harga MDF per lembar) cost_electronics = 350000; % Rp (Arduino, motor, sensor) cost_other = 100000; % Rp (baut, kabel, dll) total_cost = cost_mdf + cost_electronics + cost_other; --- Informasi Gundukan --- Tinggi gundukan : 20 mm Lebar gundukan : 40 mm (20 x 2) Jumlah gundukan : 3 buah Total lintasan : 120 mm (40 x 3) Jarak antar : 1440 mm ======================================================================== 3. fprintf(' Massa Air (m_water) : %.3f kg (%.1f ml)\n', m_water, total_water_vol); fprintf(' Massa Total (m_total) : %.3f kg\n', m_total); fprintf(' Rasio Berat/Payload : %.2f : 1\n', weight_to_payload_ratio); fprintf(' Kecepatan Rata-rata : %.2f m/s\n', avg_velocity); fprintf(' Estimasi Waktu Misi : %.1f detik\n', total_mission_time); fprintf(' Estimasi Biaya : Rp %.0f\n\n', total_cost); fprintf('--- STATUS KEPATUHAN ---\n'); % -------------------- DEMAND 1: Volume Air -------------------- if total_water_vol >= target_water_volume fprintf(' [PASS] DEMAND 1: Volume Air (%.1f ml) >= %.0f ml\n', ... total_water_vol, target_water_volume); else fprintf(' [WARN] WISHES 1: Rasio Berat (%.2f) >= %.0f:1\n', ... weight_to_payload_ratio, target_weight_ratio); end % -------------------- WISHES 3: Waktu Misi -------------------- if total_mission_time < target_mission_time fprintf(' [PASS] WISHES 3: Waktu Misi (%.1f s) < %.0f s\n', ... total_mission_time, target_mission_time); else fprintf(' [WARN] WISHES 3: Waktu Misi (%.1f s) >= %.0f s\n', ... total_mission_time, target_mission_time); end % -------------------- DEMAND 2: Stabilitas -------------------- if all(stability_status) fprintf(' [PASS] DEMAND 2: Stabilitas Terjaga (y_max < %.1f cm)\n', ... target_stability); else idx_aman = find(stability_status, 1, 'last'); if ~isempty(idx_aman) && idx_aman < length(v_test) fprintf(' [WARN] DEMAND 2: Risiko Tumpah pada Kecepatan > %.1f m/s\n', ... v_test(idx_aman)); elseif isempty(idx_aman) fprintf(' [FAIL] DEMAND 2: Tidak aman pada semua kecepatan\n'); else fprintf(' [WARN] DEMAND 2: Semua kecepatan aman\n'); end end fprintf('\n========================================\n'); ======================================== KELOMPOK 2 - DESIGN EVALUATION REPORT PRD FTMD 2026 - Kelas 56 Automatic Water Trucking Machine ======================================== --- PARAMETER KENDARAAN --- Massa Kering (m_dry) : 1.450 kg Massa Air (m_water) : 0.680 kg (680.0 ml) Massa Total (m_total) : 2.130 kg Rasio Berat/Payload : 2.13 : 1 Kecepatan Rata-rata : 0.15 m/s Estimasi Waktu Misi : 50.0 detik Estimasi Biaya : Rp 500000 --- STATUS KEPATUHAN --- [PASS] DEMAND 1: Volume Air (680.0 ml) >= 550 ml [WARN] WISHES 1: Rasio Berat (2.13) >= 2:1 [PASS] WISHES 3: Waktu Misi (50.0 s) < 60 s [WARN] DEMAND 2: Semua kecepatan aman ======================================== ======================================================================== 4.
